layout (location = 0) in vec4 aPos;
layout (location = 1) in vec4 aColor;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

out vec4 vColor;

void main()
{
    gl_Position = projection * view * model * aPos;
    vColor = aColor;
}
